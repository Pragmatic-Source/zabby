def initialize(zname, zmethods)
  @zname = zname
  @zmethods = zmethods.map { |f| f.to_sym }
end

def method_missing(zmethod, *args, &block)
  if @zmethods.include? zmethod
    Zabby::Runner.instance.connection.perform_request(@zname, zmethod, args.first)
  else
    super
  end
end
